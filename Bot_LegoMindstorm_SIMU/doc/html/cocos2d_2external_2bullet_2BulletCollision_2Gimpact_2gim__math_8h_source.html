119 #define GIM_INV_SQRT(va,isva)\
120 {\
121  if(va<=0.0000001f)\
122  {\
123  isva = G_REAL_INFINITY;\
124  }\
125  else\
126  {\
127  GREAL _x = va * 0.5f;\
128  GUINT _y = 0x5f3759df - ( GIM_IR(va) >> 1);\
129  isva = GIM_FR(_y);\
130  isva = isva * ( 1.5f - ( _x * isva * isva ) );\
