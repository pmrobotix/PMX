95 template <typename T, typename O = ObjectTraits<T>, typename locking_traits = lockless_semantics>
96 class AllocatorStrategyPool
97  : public AllocatorStrategyFixedBlock<sizeof(T), O::alignment, locking_traits>
98  , public O
99 {
100 public:
101 
102  typedef T value_type;
103  typedef value_type* pointer;
104 
106  typedef AllocatorStrategyFixedBlock<sizeof(T), O::alignment, locking_traits> tParentStrategy;
107 
108  AllocatorStrategyPool(const char* tag = nullptr, size_t poolSize = 100)
109  : tParentStrategy(tag)
110  {
111  poolSize = Configuration::getInstance()->getValue(tag, Value((int)poolSize)).asInt();
112  tParentStrategy::_pageSize = poolSize;
113  }
114 
121  CC_ALLOCATOR_INLINE void* allocate(size_t size)
122  {
123  T* object;
124  if (sizeof(T) == size)
125  {
126  object = (pointer)tParentStrategy::allocate(sizeof(T));
127  }
128  else
129  {
130  object = (T*)ccAllocatorGlobal.allocate(size);
131  }
132  O::construct(object);
133  return object;
134  }
135 
142  CC_ALLOCATOR_INLINE void deallocate(void* address, size_t size = 0)
143  {
144  if (address)
145  {
146  O::destroy((T*)address);
147  if (sizeof(T) == size)
148  {
149  tParentStrategy::deallocate(address, sizeof(T));
150  }
151  else
152  {
153  ccAllocatorGlobal.deallocate(address, size);
154  }
155  }
156  }
157 
158 #if CC_ENABLE_ALLOCATOR_DIAGNOSTICS
159  std::string diagnostics() const

162  s << AllocatorBase::tag() << " initial:" << tParentStrategy::_pageSize << " count:" << tParentStrategy::_allocated << " highest:" << tParentStrategy::_highestCount << "\n";
163  return s.str();
164  }
